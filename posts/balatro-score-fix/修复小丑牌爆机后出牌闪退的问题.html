修复小丑牌爆机后出牌闪退的问题

-游戏 -Lua

最近玩小丑牌有点上头，想打个无尽39底注爆机玩玩。结果在17底注左右的时候不小心打爆了一次，发现后续一旦出牌就直接闪退，查了一下发现是一个Bug，遂尝试修复

给不知道的人科普一下：小丑牌的分数数据使用double类型存储，一旦获得的分数超出了double能存储的最大数值(1.797e308)，这次出牌的得分就会显示为nan

而在小丑牌的无尽模式中，筹码所需要的分数是成指数型增长的，到了39底注时，回合的至少得分也超出了这个数据上限，正常情况下无法打败，某种意义上算是无尽的终点

读档后第一次正常出牌会在这里做 正常分数 > 坏的旧记录 比较，然后游戏闪退

lua
function check_and_set_high_score(score, amt)
  if not amt or type(amt) ~= 'number' then return end
  if G.GAME.round_scores[score] and math.floor(amt) > G.GAME.round_scores[score].amt then
    G.GAME.round_scores[score].amt = math.floor(amt)
  end
  if  G.GAME.seeded  then return end
  if score == 'hand' and G.SETTINGS.COMP and ((not G.SETTINGS.COMP.score) or (G.SETTINGS.COMP.score < math.floor(amt))) then 
    G.SETTINGS.COMP.score = amt
    send_score(math.floor(amt))
  end
  if G.PROFILES[G.SETTINGS.profile].high_scores[score] and math.floor(amt) > G.PROFILES[G.SETTINGS.profile].high_scores[score].amt then
    if G.GAME.round_scores[score] then G.GAME.round_scores[score].high_score = true end
    G.PROFILES[G.SETTINGS.profile].high_scores[score].amt = math.floor(amt)
    G:save_settings()
  end
end
lua
function check_and_set_high_score(score, amt)
  if not amt or type(amt) ~= 'number' or amt ~= amt then return end
  local amt_floor = math.floor(amt)
  if G.GAME.round_scores[score] then
    local round_amt = G.GAME.round_scores[score].amt
    if type(round_amt) ~= 'number' or round_amt ~= round_amt then round_amt = 0 end
    if amt_floor > round_amt then
      G.GAME.round_scores[score].amt = amt_floor
    end
  end
  if  G.GAME.seeded  then return end
  if score == 'hand' and G.SETTINGS.COMP and ((not G.SETTINGS.COMP.score) or (G.SETTINGS.COMP.score < amt_floor)) then 
    G.SETTINGS.COMP.score = amt
    send_score(amt_floor)
  end
  if G.PROFILES[G.SETTINGS.profile].high_scores[score] then
    local high_score_amt = G.PROFILES[G.SETTINGS.profile].high_scores[score].amt
    if type(high_score_amt) ~= 'number' or high_score_amt ~= high_score_amt then high_score_amt = 0 end
    if amt_floor > high_score_amt then
    if G.GAME.round_scores[score] then G.GAME.round_scores[score].high_score = true end
      G.PROFILES[G.SETTINGS.profile].high_scores[score].amt = amt_floor
    G:save_settings()
  end
  end
end
